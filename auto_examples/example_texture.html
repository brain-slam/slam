    import numpy as np
from slam import texture
from slam import io as sio

    tex = sio.load_texture("../examples/data/example_texture.gii")
print(tex)
print(tex.metadata)
print(tex.shape)
print(tex.dtype)
print(tex.min())
print(tex.max())
